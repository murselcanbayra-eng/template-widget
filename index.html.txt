  document.documentElement.setAttribute('data-theme', t);

}

  try {
    const t = localStorage.getItem('yks_theme') || 'dark';
    setTheme(t);

// ═══════════════════ STORAGE ═══════════════════
const STORAGE_KEYS = ['study-logs', 'nets', 'konu-done', 'program-done', 'program-custom'];
const _cache = {};

async function storageLoad() {
  for (const key of STORAGE_KEYS) {
    try {
      const res = await window.storage.get(key);
      if (res && res.value) {
        _cache[key] = JSON.parse(res.value);
      } else {
        _cache[key] = (key === 'konu-done' || key === 'program-done' || key === 'program-custom') ? {} : [];
      }
    } catch(e) {
      _cache[key] = (key === 'konu-done' || key === 'program-done' || key === 'program-custom') ? {} : [];
    }
  }
}

async function storageSave(key, data) {
  _cache[key] = data;
  try {
    await window.storage.set(key, JSON.stringify(data));
  } catch(e) {
    try { localStorage.setItem('yks_' + key, JSON.stringify(data)); } catch(e) {}
  }
}

function load(key) {
  if (_cache[key] !== undefined) return Array.isArray(_cache[key]) ? _cache[key] : [];
  try { return JSON.parse(localStorage.getItem('yks_' + key)) || []; } catch(e) { return []; }
}
function save(key, data) { _cache[key] = data; storageSave(key, data); }
function loadObj(key, def) {
  if (_cache[key] !== undefined) return (typeof _cache[key] === 'object' && !Array.isArray(_cache[key])) ? _cache[key] : def;
  try { return JSON.parse(localStorage.getItem('yks_' + key)) || def; } catch(e) { return def; }
}

// ═══════════════════ UTILS ═══════════════════
function fmtTime(seconds) {
  const h = Math.floor(seconds / 3600);
  const m = Math.floor((seconds % 3600) / 60);
  const s = seconds % 60;
  if (h > 0) return `${h}s ${m}dk`;
  if (m > 0) return `${m}dk`;
  return `${s}sn`;
}
function fmtHMS(seconds) {
  const h = String(Math.floor(seconds / 3600)).padStart(2, '0');
  const m = String(Math.floor((seconds % 3600) / 60)).padStart(2, '0');
  const s = String(seconds % 60).padStart(2, '0');
  return `${h}:${m}:${s}`;
}
function today() { return new Date().toISOString().split('T')[0]; }
function getSubject(id) { return SUBJECTS.find(s => s.id === id); }
function toast(msg, type = 'success') {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.style.background = type === 'success' ? 'var(--green)' : type === 'error' ? 'var(--red)' : 'var(--accent)';
  t.classList.add('show');
  setTimeout(() => t.classList.remove('show'), 2500);
}
function daysUntil(dateStr) {
  const target = new Date(dateStr);
  const now = new Date();
  now.setHours(0,0,0,0);
  return Math.ceil((target - now) / (1000*60*60*24));
}

function posChartTip(e) {
  const tip = document.getElementById('chart-floating-tooltip');
  if (!tip || tip.style.display === 'none') return;
  const tw = tip.offsetWidth, th = tip.offsetHeight;
  const vw = window.innerWidth, vh = window.innerHeight;
  let x = e.clientX + 12;
  let y = e.clientY - th - 12;
  if (x + tw > vw - 8) x = e.clientX - tw - 12;
  if (y < 8) y = e.clientY + 16;
  tip.style.left = x + 'px';
  tip.style.top = y + 'px';
}

  const tip = document.getElementById('chart-floating-tooltip');
  if (tip) tip.style.display = 'none';
}

function updateNetPreview(fid, max) {
  const fields = document.getElementById('net-type').value === 'tyt' ? TYT_NETS : AYT_NETS;
  const d = parseFloat(document.getElementById(`net-d-${fid}`)?.value) || 0;
  const y = parseFloat(document.getElementById(`net-y-${fid}`)?.value) || 0;
  const net = calcNet(d, y);
  const pct = net / max;
  const color = pct >= 0.7 ? 'var(--green)' : pct >= 0.4 ? 'var(--yellow)' : net > 0 ? 'var(--red)' : 'var(--text3)';
  const prev = document.getElementById(`net-prev-${fid}`);
  if (prev) prev.innerHTML = `<span style="color:${color}">Net: ${net.toFixed(2)}</span>`;

  // Toplam net güncelle
  let total = 0;
  fields.forEach(f => {
    const dd = parseFloat(document.getElementById(`net-d-${f.id}`)?.value) || 0;
    const yy = parseFloat(document.getElementById(`net-y-${f.id}`)?.value) || 0;
    total += calcNet(dd, yy);
  });
  const tp = document.getElementById('net-total-preview');
  if (tp) tp.textContent = total.toFixed(2);
}

function addNet() {
  const type = document.getElementById('net-type').value;
  const date = document.getElementById('net-date').value || today();
  const name = document.getElementById('net-name').value.trim() || 'Deneme';
  const fields = type === 'tyt' ? TYT_NETS : AYT_NETS;
  const scores = {};
  const raw = {}; // doğru/yanlış ham veriler
  fields.forEach(f => {
    const d = parseFloat(document.getElementById('net-d-' + f.id)?.value) || 0;
    const y = parseFloat(document.getElementById('net-y-' + f.id)?.value) || 0;
    scores[f.id] = calcNet(d, y);
    raw[f.id] = { d, y };
  });
  const total = Object.values(scores).reduce((a, b) => a + b, 0);
  const nets = load('nets');
  nets.unshift({ id: Date.now(), type, date, name, scores, raw, total });
  save('nets', nets);
  document.getElementById('net-name').value = '';
  fields.forEach(f => {
    const de = document.getElementById('net-d-' + f.id);
    const ye = document.getElementById('net-y-' + f.id);
    if (de) de.value = '';
    if (ye) ye.value = '';
    const prev = document.getElementById('net-prev-' + f.id);
    if (prev) prev.innerHTML = 'Net: —';
  });
  const tp = document.getElementById('net-total-preview');
  if (tp) tp.textContent = '—';
  toast('✅ Deneme kaydedildi!');
  renderNets();
}

function getLessonColor(name) {
  let hash = 0;
  for (let i = 0; i < name.length; i++) hash = name.charCodeAt(i) + ((hash << 5) - hash);
  return LESSON_COLORS[Math.abs(hash) % LESSON_COLORS.length];
}

function getCustomProgram() {
  const stored = loadObj('program-custom', {});
  // Eğer hiç özelleştirilmemişse varsayılan programı döndür
  if (!Object.keys(stored).length) {
    const def = {};
    DAYS.forEach(d => {
      def[d] = (WEEKLY_PROGRAM[d] || []).map(l => ({ name: l.name, time: l.time }));
    });
    return def;
  }
  return stored;
}

function saveCustomProgram(prog) {
  save('program-custom', prog);
}

function getTodayName() {
  const names = ['Pazar', 'Pazartesi', 'Salı', 'Çarşamba', 'Perşembe', 'Cuma', 'Cumartesi'];
  return names[new Date().getDay()];
}

function addProgLesson() {
  const day = document.getElementById('prog-day').value;
  const name = document.getElementById('prog-name').value.trim();
  const time = document.getElementById('prog-time').value.trim() || 'Gün içi';
  if (!name) { toast('Ders adı gir!', 'error'); return; }
  const prog = getCustomProgram();
  if (!prog[day]) prog[day] = [];
  prog[day].push({ name, time });
  saveCustomProgram(prog);
  document.getElementById('prog-name').value = '';
  document.getElementById('prog-time').value = '';
  toast('✅ Ders eklendi!');
  renderProgram();
}

function deleteProgLesson(day, idx) {
  const prog = getCustomProgram();
  if (!prog[day]) return;
  prog[day].splice(idx, 1);
  // Tamamlandı işaretlerini de temizle
  const done = loadObj('program-done', {});
  // Shift keys after removed index
  const newDone = {};
  Object.keys(done).forEach(k => {
    if (!k.startsWith(day + '-')) { newDone[k] = done[k]; return; }
    const i = parseInt(k.split('-').pop());
    if (i < idx) newDone[k] = done[k];
    else if (i > idx) newDone[`${day}-${i-1}`] = done[k];
  });
  save('program-done', newDone);
  saveCustomProgram(prog);
  renderProgram();
}

function renderProgram() {
  const todayName = getTodayName();
  const done = loadObj('program-done', {});
  const prog = getCustomProgram();
  const grid = document.getElementById('program-grid');

  grid.innerHTML = DAYS.map(day => {
    const lessons = prog[day] || [];
    const isToday = day === todayName;
    return `<div class="day-card ${isToday ? 'today' : ''}">
      <div class="day-name">${day}${isToday ? ' 📍' : ''}</div>
      ${lessons.length === 0
        ? `<div style="font-size:11px;color:var(--text3);text-align:center;padding:8px 0">Ders yok</div>`
        : lessons.map((l, i) => {
            const key = `${day}-${i}`;
            const isDone = done[key];
            const col = getLessonColor(l.name);
            return `<div style="
                font-size:12px; font-weight:600; padding:5px 8px; border-radius:6px;
                margin-bottom:5px; cursor:pointer; display:flex; align-items:center; gap:6px;
                background:${col}22; color:${isDone ? 'var(--text3)' : col};
                ${isDone ? 'text-decoration:line-through; opacity:.5;' : ''}
                border:1px solid ${col}33; transition:all .2s;
              "
              onclick="toggleProgramDone('${key}')"
              title="${l.time}">
              <span style="flex:1">${isDone ? '✓ ' : ''}${l.name}</span>
              <span onclick="event.stopPropagation(); deleteProgLesson('${day}',${i})"
                style="font-size:10px; opacity:.5; cursor:pointer; padding:0 2px; border-radius:3px;"
                onmouseenter="this.style.opacity='1'; this.style.color='var(--red)'"
                onmouseleave="this.style.opacity='.5'; this.style.color=''">✕</span>
            </div>`;
          }).join('')
      }
    </div>`;
  }).join('');

  // Bugünün görevleri
  const todayLessons = prog[todayName] || [];
  const todayEl = document.getElementById('today-tasks');
  if (!todayLessons.length) {
    todayEl.innerHTML = '<div class="empty">Bugün programda ders yok 🎉</div>';
    return;
  }
  todayEl.innerHTML = todayLessons.map((l, i) => {
    const key = `${todayName}-${i}`;
    const isDone = done[key];
    const col = getLessonColor(l.name);
    return `<div class="log-item" style="cursor:pointer; ${isDone ? 'opacity:.5;' : ''}"
      onclick="toggleProgramDone('${key}')">
      <div class="log-subject-dot" style="background:${isDone ? 'var(--green)' : col}"></div>
      <div class="log-info">
        <div class="log-subject" style="color:${col}">${isDone ? '✓ ' : ''}${l.name}</div>
        <div class="log-topic">${l.time}</div>
      </div>
      <div style="font-size:20px">${isDone ? '✅' : '⭕'}</div>
    </div>`;
  }).join('');
}

function toggleProgramDone(key) {
  const done = loadObj('program-done', {});
  done[key] = !done[key];
  save('program-done', done);
  renderProgram();
}

// ═══════════════════ DASHBOARD ═══════════════════
function renderDashboard() {
  const logs = load('study-logs');
  const nets = load('nets');

  const totalSeconds = logs.reduce((a, l) => a + l.seconds, 0);
  document.getElementById('dash-total-time').textContent = fmtTime(totalSeconds) || '0s';

  const weekAgo = new Date(); weekAgo.setDate(weekAgo.getDate() - 7);
  const weekLogs = logs.filter(l => new Date(l.date) >= weekAgo);
  const weekSeconds = weekLogs.reduce((a, l) => a + l.seconds, 0);
  document.getElementById('dash-week-time').textContent = fmtTime(weekSeconds) || '0s';
  document.getElementById('dash-deneme-count').textContent = nets.length;

  const subjectTimes = {};
  logs.forEach(l => { subjectTimes[l.subject] = (subjectTimes[l.subject] || 0) + l.seconds; });
  const maxTime = Math.max(...Object.values(subjectTimes), 1);
  const progEl = document.getElementById('dash-subject-progress');
  const sorted = SUBJECTS.filter(s => subjectTimes[s.id] > 0).sort((a, b) => (subjectTimes[b.id] || 0) - (subjectTimes[a.id] || 0));
  if (!sorted.length) { progEl.innerHTML = '<div class="empty">Henüz çalışma kaydı yok</div>'; }
  else progEl.innerHTML = sorted.slice(0, 8).map(s => {
    const secs = subjectTimes[s.id] || 0;
    const pct = (secs / maxTime) * 100;
    return `<div class="prog-item">
      <div class="prog-header">
        <span class="prog-label" style="color:${s.color}">${s.name}</span>
        <span class="prog-val" style="color:${s.color}">${fmtTime(secs)}</span>
      </div>
      <div class="prog-bar"><div class="prog-fill" style="width:${pct}%; background:${s.color}"></div></div>
    </div>`;
  }).join('');

  const weekEl = document.getElementById('dash-week-summary');
  const wBySubject = {};
  weekLogs.forEach(l => { wBySubject[l.subject] = (wBySubject[l.subject] || 0) + l.seconds; });
  const topSubjects = SUBJECTS.filter(s => wBySubject[s.id] > 0);
  if (!topSubjects.length) { weekEl.innerHTML = '<div class="empty" style="grid-column:1/-1">Bu hafta çalışma yok</div>'; }
  else weekEl.innerHTML = topSubjects.map(s => `
    <div class="week-subject-card">
      <div class="week-subject-name" style="color:${s.color}">${s.name.replace('TYT ','').replace('AYT ','')}</div>
      <div class="week-subject-time" style="color:${s.color}">${fmtTime(wBySubject[s.id] || 0)}</div>
    </div>
  `).join('');

  const chartEl = document.getElementById('dash-activity-chart');
  const labelEl = document.getElementById('dash-activity-labels');
  const scrollEl = document.getElementById('dash-activity-scroll');
  const countEl = document.getElementById('dash-activity-count');
  const allDates = logs.map(l => l.date).filter(Boolean);
  const today_str = today();
  let startDate;
  if (allDates.length === 0) {
    const d = new Date(); d.setDate(d.getDate() - 13);
    startDate = d.toISOString().split('T')[0];
  } else {
    startDate = allDates.slice().sort()[0];
  }
  const allDays = [];
  const cur = new Date(startDate);
  const end = new Date(today_str);
  while (cur <= end) {
    allDays.push(cur.toISOString().split('T')[0]);
    cur.setDate(cur.getDate() + 1);
  }
  const dayTotals = allDays.map(d => logs.filter(l => l.date === d).reduce((a, l) => a + l.seconds, 0));
  const maxD = Math.max(...dayTotals, 1);
  const activeDays = dayTotals.filter(t => t > 0).length;
  if (countEl) countEl.textContent = `${allDays.length} gün · ${activeDays} aktif`;

  chartEl.innerHTML = dayTotals.map((t, i) => {
    const h = Math.max(4, (t / maxD) * 96);
    const hrs = Math.floor(t / 3600);
    const mins = Math.floor((t % 3600) / 60);
    const label = t > 0 ? (hrs > 0 ? `${hrs}s ${mins}dk` : `${mins}dk`) : '';
    const isToday = allDays[i] === today_str;
    return `<div class="chart-col" style="justify-content:flex-end;" data-tip="${allDays[i]}&#10;${label || 'Çalışma yok'}" onmouseenter="showChartTip(event,this)" onmouseleave="hideChartTip()">
      <div style="width:20px; height:${h}px; border-radius:4px 4px 0 0;
        background:linear-gradient(to top, var(--accent), var(--accent2));
        opacity:${t>0?0.85:0.15}; flex-shrink:0;
        ${isToday ? 'box-shadow:0 0 6px var(--accent2);' : ''}"></div>
    </div>`;
  }).join('');

  const dayNames = ['Paz', 'Pzt', 'Sal', 'Çar', 'Per', 'Cum', 'Cmt'];
  labelEl.innerHTML = dayTotals.map((t, i) => {
    const d = new Date(allDays[i]);
    const isToday = allDays[i] === today_str;
    const isMon = d.getDay() === 1;
    const lbl = isToday ? '<span style="color:var(--accent2)">Bugün</span>' : (isMon ? dayNames[d.getDay()] + '<br><span style="font-size:9px">' + allDays[i].slice(5) + '</span>' : '');
    return `<div style="width:28px; min-width:28px; text-align:center; font-size:9px; color:var(--text3); line-height:1.3">${lbl}</div>`;
  }).join('');
  setTimeout(() => { if (scrollEl) scrollEl.scrollLeft = scrollEl.scrollWidth; }, 50);

  renderNetChart();
  renderDashboardProgress();
}

let ktActiveTab = 'tyt';
let aytProgAlan = 'ayt_say';

function ktShowTab(tab, btn) {
  ktActiveTab = tab;
  document.querySelectorAll('.kt-tab').forEach(t => t.classList.remove('active'));
  if (btn) btn.classList.add('active');
  renderKonular();
}
function toggleKonu(dersId, konuIdx) {
  // AYT konuları için key'e alan adı dahil
  const isAyt = ktActiveTab.startsWith('ayt_');
  const key = isAyt ? `konu-${dersId}-${ktActiveTab}-${konuIdx}` : `konu-${dersId}-${konuIdx}`;
  const done = loadObj('konu-done', {});
  done[key] = !done[key];
  save('konu-done', done);
  renderKonular();
  renderDashboardProgress();
}
function renderKonular() {
  const done = loadObj('konu-done', {});
  const dersler = KONU_DATA[ktActiveTab];
  const isAyt = ktActiveTab.startsWith('ayt_');
  const el = document.getElementById('kt-content');
  const logs = load('study-logs');
  const qMap = {};
  logs.forEach(l => {
    if (!l.questions || !l.topic || l.topic === '—') return;
    if (!qMap[l.subject]) qMap[l.subject] = {};
    qMap[l.subject][l.topic] = (qMap[l.subject][l.topic] || 0) + l.questions;
  });
  el.innerHTML = dersler.map(ders => {
    const total = ders.konular.length;
    const tamamlanan = ders.konular.filter((k, i) => {
      const key = isAyt ? `konu-${ders.id}-${ktActiveTab}-${i}` : `konu-${ders.id}-${i}`;
      return done[key];
    }).length;
    const pct = total > 0 ? Math.round((tamamlanan / total) * 100) : 0;
    const dersQ = qMap[ders.id] || {};
    return `<div class="kt-ders-block card">
      <div class="kt-ders-header">
        <span style="width:12px;height:12px;border-radius:3px;background:${ders.color};display:inline-block;flex-shrink:0"></span>
        <span class="kt-ders-name" style="color:${ders.color}">${ders.ders}</span>
        <span class="kt-ders-progress">${tamamlanan}/${total} (${pct}%)</span>
      </div>
      <div class="kt-ders-bar"><div class="kt-ders-bar-fill" style="width:${pct}%; background:${ders.color}"></div></div>
      <div class="kt-konular">
        ${ders.konular.map((k, i) => {
          const key = isAyt ? `konu-${ders.id}-${ktActiveTab}-${i}` : `konu-${ders.id}-${i}`;
          const isDone = done[key];
          const qCount = dersQ[k.ad] || 0;
          const qBadge = qCount > 0
            ? `<span style="font-family:'DM Mono',monospace;font-size:11px;font-weight:700;color:var(--accent2);background:color-mix(in srgb, var(--accent2) 12%, transparent);border-radius:6px;padding:1px 6px;flex-shrink:0">${qCount}s</span>`
            : '';
          return `<div class="kt-konu ${isDone ? 'done' : ''}" onclick="toggleKonu('${ders.id}', ${i})">
            <div class="kt-checkbox">${isDone ? '✓' : ''}</div>
            <span class="kt-konu-name">${k.ad}</span>
            ${qBadge}
          </div>`;
        }).join('')}
      </div>
    </div>`;
  }).join('');
}
// ═══════════════════ NET CHART ═══════════════════
let netChartType = 'tyt';

function setNetChartType(type, btn) {
  netChartType = type;
  document.querySelectorAll('.net-chart-btn').forEach(b => {
    b.style.background = 'var(--surface2)';
    b.style.color = 'var(--text3)';
    b.style.borderColor = 'var(--border)';
  });
  if (btn) { btn.style.background = 'var(--accent)'; btn.style.color = 'white'; btn.style.borderColor = 'var(--accent)'; }
  renderNetChart();
}

function renderNetChart() {
  const allNets = load('nets');
  const nets = allNets.filter(n => n.type === netChartType).slice().reverse(); // eskiden yeniye
  const svg = document.getElementById('dash-net-svg');
  const empty = document.getElementById('dash-net-empty');
  const wrap = document.getElementById('dash-net-chart-wrap');
  const tooltip = document.getElementById('dash-net-tooltip');
  if (!svg) return;

  if (!nets.length) {
    svg.style.display = 'none';
    if (empty) empty.style.display = 'block';
    return;
  }
  svg.style.display = 'block';
  if (empty) empty.style.display = 'none';

  const fields = netChartType === 'tyt' ? TYT_NETS : AYT_NETS;
  const maxPossible = fields.reduce((a, f) => a + f.max, 0);

  const W = svg.parentElement.clientWidth || 600;
  const H = 220;
  const PAD = { top: 24, right: 28, bottom: 48, left: 48 };
  const cW = W - PAD.left - PAD.right;
  const cH = H - PAD.top - PAD.bottom;

  svg.setAttribute('viewBox', `0 0 ${W} ${H}`);
  svg.setAttribute('height', H);

  const vals = nets.map(n => n.total);
  const minVal = Math.max(0, Math.min(...vals) - 5);
  const maxVal = Math.min(maxPossible, Math.max(...vals) + 5);
  const range = maxVal - minVal || 1;

  const xStep = nets.length > 1 ? cW / (nets.length - 1) : cW / 2;
  const px = (i) => PAD.left + (nets.length > 1 ? i * xStep : cW / 2);
  const py = (v) => PAD.top + cH - ((v - minVal) / range) * cH;

  // Renk gradyanı aksen rengiyle
  const accentColor = netChartType === 'tyt' ? '#4d9fff' : '#6fcf8e';
  const accentLight = netChartType === 'tyt' ? 'rgba(77,159,255,' : 'rgba(111,207,142,';

  // Y eksen grid çizgileri
  const gridCount = 5;
  let gridLines = '';
  for (let i = 0; i <= gridCount; i++) {
    const v = minVal + (range / gridCount) * i;
    const y = py(v);
    gridLines += `<line x1="${PAD.left}" y1="${y}" x2="${W - PAD.right}" y2="${y}"
      stroke="var(--border)" stroke-width="1" stroke-dasharray="4,4" opacity="0.6"/>
      <text x="${PAD.left - 6}" y="${y + 4}" text-anchor="end" font-size="10"
        font-family="DM Mono,monospace" fill="var(--text3)">${v.toFixed(0)}</text>`;
  }

  // Alan doldurma (gradient)
  let areaPath = `M ${px(0)} ${py(vals[0])}`;
  for (let i = 1; i < nets.length; i++) {
    // Bezier eğrisi için kontrol noktaları
    const x0 = px(i - 1), y0 = py(vals[i - 1]);
    const x1 = px(i), y1 = py(vals[i]);
    const cx = (x0 + x1) / 2;
    areaPath += ` C ${cx} ${y0} ${cx} ${y1} ${x1} ${y1}`;
  }
  const areaFill = areaPath +
    ` L ${px(nets.length - 1)} ${PAD.top + cH}` +
    ` L ${px(0)} ${PAD.top + cH} Z`;

  // Çizgi
  let linePath = `M ${px(0)} ${py(vals[0])}`;
  for (let i = 1; i < nets.length; i++) {
    const x0 = px(i - 1), y0 = py(vals[i - 1]);
    const x1 = px(i), y1 = py(vals[i]);
    const cx = (x0 + x1) / 2;
    linePath += ` C ${cx} ${y0} ${cx} ${y1} ${x1} ${y1}`;
  }

  // X ekseni etiketleri + noktalar
  let xLabels = '';
  let dots = '';
  nets.forEach((n, i) => {
    const x = px(i), y = py(n.total);
    const shortDate = n.date.slice(5); // MM-DD
    const showLabel = nets.length <= 12 || i % Math.ceil(nets.length / 10) === 0 || i === nets.length - 1;
    if (showLabel) {
      xLabels += `<text x="${x}" y="${H - 6}" text-anchor="middle" font-size="10"
        font-family="DM Mono,monospace" fill="var(--text3)">${shortDate}</text>
        <text x="${x}" y="${H - 18}" text-anchor="middle" font-size="9"
        font-family="Plus Jakarta Sans,sans-serif" fill="var(--text3)"
        style="overflow:hidden">${n.name.length > 8 ? n.name.slice(0,8)+'…' : n.name}</text>`;
    }
    // Renk noktaya göre: yeşil/sarı/kırmızı
    const pct = n.total / maxPossible * 100;
    const dotColor = pct >= 65 ? '#34d399' : pct >= 40 ? '#fbbf24' : '#f87171';
    dots += `<circle class="net-chart-dot" cx="${x}" cy="${y}" r="5"
      fill="${dotColor}" stroke="var(--bg)" stroke-width="2"
      data-idx="${i}"
      onmouseenter="showNetTip(event,${i})"
      onmouseleave="hideNetTip()"/>`;
  });

  svg.innerHTML = `
    <defs>
      <linearGradient id="netAreaGrad" x1="0" y1="0" x2="0" y2="1">
        <stop offset="0%" stop-color="${accentColor}" stop-opacity="0.25"/>
        <stop offset="100%" stop-color="${accentColor}" stop-opacity="0.02"/>
      </linearGradient>
    </defs>
    ${gridLines}
    <path d="${areaFill}" fill="url(#netAreaGrad)"/>
    <path d="${linePath}" fill="none" stroke="${accentColor}" stroke-width="2.5" stroke-linecap="round" stroke-linejoin="round"/>
    ${xLabels}
    ${dots}
  `;

  // Tooltip mantığı
  svg._nets = nets;
  svg._maxPossible = maxPossible;
  svg._fields = fields;
}

function showNetTip(e, idx) {
  const svg = document.getElementById('dash-net-svg');
  const tip = document.getElementById('dash-net-tooltip');
  if (!svg || !tip || !svg._nets) return;
  const n = svg._nets[idx];
  const fields = svg._fields;
  const maxTotal = svg._maxPossible;
  const pct = (n.total / maxTotal * 100).toFixed(1);

  const pills = fields.map(f => {
    const net = (n.scores[f.id] || 0).toFixed(2);
    const raw = n.raw?.[f.id];
    const rawStr = raw ? ` (${raw.d}D/${raw.y}Y)` : '';
    return `${f.label}: <b>${net}</b>${rawStr}`;
  }).join('<br>');

  tip.innerHTML = `<div style="font-size:13px;font-weight:800;color:var(--text);margin-bottom:4px">${n.name}</div>
    <div style="font-size:11px;color:var(--text3);margin-bottom:6px">${n.date} · ${n.type.toUpperCase()}</div>
    <div style="font-size:11px;color:var(--text2);margin-bottom:6px;line-height:1.9">${pills}</div>
    <div style="font-size:15px;font-weight:800;color:var(--accent)">Toplam: ${n.total.toFixed(2)} / ${maxTotal} <span style="color:var(--text3);font-size:11px">(${pct}%)</span></div>`;

  tip.style.display = 'block';
  const rect = svg.getBoundingClientRect();
  const wrapRect = document.getElementById('dash-net-chart-wrap').getBoundingClientRect();
  let left = e.clientX - wrapRect.left + 12;
  let top = e.clientY - wrapRect.top - 20;
  if (left + 260 > wrapRect.width) left = e.clientX - wrapRect.left - 270;
  if (top < 0) top = 8;
  tip.style.left = left + 'px';
  tip.style.top = top + 'px';
}

function hideNetTip() {
  const tip = document.getElementById('dash-net-tooltip');
  if (tip) tip.style.display = 'none';
}

      const tamamlanan = ders.konular.filter((k, i) => done[`konu-${ders.id}-${i}`]).length;
      totalAll += total; doneAll += tamamlanan;
      const pct = total > 0 ? Math.round((tamamlanan / total) * 100) : 0;
      return `<div class="ders-prog-row">
        <span class="ders-prog-name" style="color:${ders.color}">${ders.ders.replace('TYT ','')}</span>
        <div class="ders-prog-bar"><div class="ders-prog-fill" style="width:${pct}%;background:${ders.color}"></div></div>
        <span class="ders-prog-val">${tamamlanan}/${total}</span>
      </div>`;
    }).join('');
    const totalPct = totalAll > 0 ? Math.round((doneAll / totalAll) * 100) : 0;
    const offset = 175.9 - (totalPct / 100) * 175.9;
    const ring = document.getElementById('ring-tyt');
    if (ring) ring.style.strokeDashoffset = offset;
    const pctEl = document.getElementById('ring-tyt-pct');
    if (pctEl) pctEl.textContent = totalPct + '%';
    const labelEl = document.getElementById('ring-tyt-label');
    if (labelEl) labelEl.textContent = `${doneAll} / ${totalAll} konu tamamlandı`;
    const rowsEl = document.getElementById('tyt-ders-progress');
    if (rowsEl) rowsEl.innerHTML = dersRows;
  }

  // AYT — seçili alan
  {
    const dersler = KONU_DATA[aytProgAlan] || [];
    let totalAll = 0, doneAll = 0;
    const dersRows = dersler.map(ders => {
      const total = ders.konular.length;
      const tamamlanan = ders.konular.filter((k, i) => done[`konu-${ders.id}-${aytProgAlan}-${i}`]).length;
      totalAll += total; doneAll += tamamlanan;
      const pct = total > 0 ? Math.round((tamamlanan / total) * 100) : 0;
      return `<div class="ders-prog-row">
        <span class="ders-prog-name" style="color:${ders.color}">${ders.ders.replace('AYT ','')}</span>
        <div class="ders-prog-bar"><div class="ders-prog-fill" style="width:${pct}%;background:${ders.color}"></div></div>
        <span class="ders-prog-val">${tamamlanan}/${total}</span>
      </div>`;
    }).join('');
    const totalPct = totalAll > 0 ? Math.round((doneAll / totalAll) * 100) : 0;
    const offset = 175.9 - (totalPct / 100) * 175.9;
    const ring = document.getElementById('ring-ayt');
    if (ring) ring.style.strokeDashoffset = offset;
    const pctEl = document.getElementById('ring-ayt-pct');
    if (pctEl) pctEl.textContent = totalPct + '%';
    const labelEl = document.getElementById('ring-ayt-label');
    if (labelEl) labelEl.textContent = `${doneAll} / ${totalAll} konu tamamlandı`;
    const rowsEl = document.getElementById('ayt-ders-progress');
    if (rowsEl) rowsEl.innerHTML = dersRows;
  }
}

function exportData() {
  const data = {
    version: 1,
    exportDate: new Date().toISOString(),
    'study-logs': load('study-logs'),
    'nets': load('nets'),
    'konu-done': loadObj('konu-done', {}),
    'program-done': loadObj('program-done', {}),
    'program-custom': loadObj('program-custom', {}),
  };
  const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = 'yks-takip-' + today() + '.json';
  a.click();
  URL.revokeObjectURL(url);
  toast('✅ Veriler dışa aktarıldı!');
}
function importData(event) {
  const file = event.target.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = function(e) {
    try {
      const data = JSON.parse(e.target.result);
      if (!data.version) throw new Error('Geçersiz dosya');
      const keys = ['study-logs', 'nets', 'konu-done', 'program-done', 'program-custom'];
      keys.forEach(key => { if (data[key] !== undefined) save(key, data[key]); });
      renderDashboard();
      renderTodayLog();
      const active = document.querySelector('.section.active');
      const id = active?.id.replace('section-', '');
      if (id === 'log') renderLog();
      else if (id === 'netler') renderNets();
      else if (id === 'konular') renderKonular();
      else if (id === 'program') renderProgram();
      toast('✅ Veriler içe aktarıldı!');
    } catch(err) {
      toast('❌ Dosya okunamadı: ' + err.message, 'error');
    }
    event.target.value = '';
  };
  reader.readAsText(file);
}
